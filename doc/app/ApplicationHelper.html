def back_url
  ref = request.referer
  pattern = /^#{root_url}.*/
  unless pattern.match(ref).nil?
    url = ref
  else
    url = "#{root_url.chop}#{url_for :controller => params[:controller], :action => 'index'}"
  end
  
  return url
end

def link_to_wiki(content, page = nil)
  page = content if page.nil?
  link_to content, wiki_url(page), :class => 'wiki', :target => '_blank'
end

def nav(model, url_helper)
  last_page = (model.where(:world_id => @worlds).count / @limit.to_f).ceil
  cur_page = [params[:page].to_i, 1].max
  links = {}
  ordered = []

  unless params[:page].eql?(1)
    links[1] = link_to_nav(1, url_helper)
  end

  unless params[:page].eql?(last_page)
    links[last_page] = link_to_nav(last_page, url_helper)
  end

  1.upto(14) do |i|   
    unless links.size.eql?(6)
      add = (i / 2.0).ceil
      if (i % 2).eql?(1)
        add *= -1
      end
      
      links[cur_page + add] = next_link(cur_page, add, last_page, url_helper)
      links.delete(cur_page + add) if links[cur_page + add].nil?
    else
      break
    end    
  end

  links[cur_page] = cur_page   

  links.sort.each do |key, value|
    ordered.push(value)
  end

  return ordered
end

def wiki_url(page)
  "http://www.fwwiki.de/index.php/#{page.gsub(/\s?(-|\/)\s?/, '\1')}"
end
