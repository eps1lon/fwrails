def details
  @world = World.where(:short => params[:world]).first
  
  @clan = Clan.where(:world_id => @world.id, :clan_id => params[:id]).
               includes(:members, :adds, :outs).first
  
  @members = @clan.members.order("name asc")
  
  @change_msg = ['1']
end

def list
  @model = Clan
  
  @attributes = %{clan_id tag name sum_experience}
  params[:order] = 'sum_experience' unless @attributes.include?(params[:order].downcase)
  
  @clans = @model.where(:world_id => @worlds)
  @last_update = @clans.order("created_at DESC").first.created_at
  @clans = @clans.order("#{params[:order]} #{params[:by]}").offset(@offset).limit(@limit) 
  
  unless params[:name].nil?
    @clans = @clans.where("clans.name LIKE ?", "%#{params[:name]}%") 
    @model = @model.where("clans.name LIKE ?", "%#{params[:name]}%") 
  end
  
  respond_to do |format|
    format.json { render :json => @users.limit(@suggest_limit) }
    format.html { render 'clans/list'}
  end
end
