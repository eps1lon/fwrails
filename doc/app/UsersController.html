def clanchange
  @skipsearch = true
  @model = UsersClanChange
  @attributes = [
    {:db => "#{@model.table_name}.user_id", :human => "user_id"},
    {:db => "#{@model.table_name}.clan_id_old", :human => "old_clan"},
    {:db => "#{@model.table_name}.clan_id_new", :human => "new_clan"},
    {:db => "#{@model.table_name}.created_at", :human => "created_at"},
    {:db => "#{@model.table_name}.world_id", :human => "world"}
  ]
  order = @attributes.select { |attr| attr[:human] == params[:order].downcase }.first
  order ||= @attributes[3]
  params[:order] = order[:human]
  
  @users = @model.includes(:user, :world, :old_clan, :new_clan).where(:world_id => @worlds).
           order("#{order[:db]} #{params[:by]}").offset(@offset).limit(@limit) 
  
  respond_to do |format|
    format.json { render :json => @users.limit(@suggest_limit) }
    format.jpeg { render_image('jpeg', [
      {:attr => 'user_id', :left => 0},
      {:attr => 'name', :left => 0.1},
      {:attr => 'created_at', :left => 0.3},
      {:attr => 'world_id', :left => 0.9}
    ])}
    format.html {render 'users/list'}
  end
end

def details
  @world = World.includes(:language).where(:short => params[:world]).first
  if @world.nil?
    render_record_not_found and return
  end

  @user = User.includes(:clan_changes, 
                        :experience_changes,
                        :name_changes,
                        :race,
                        :race_changes,
                        :registration, 
                        :achievements).
               where(:name => params[:name], :world_id => @world).first
  if @user.nil?
    render_record_not_found and return
  end
  
  @user.world = @world
  
  # achievements
  @achievements = @user.achievements
  @achievements_progress = @user.achievement_progresses.includes(:achievement).
                                                        order("users_achievements.created_at ASC")
  # changes
  @changes = [
    @user.clan_changes.includes(:old_clan, :new_clan).order("users_clan_changes.created_at ASC"),
    @user.name_changes.select("name_old AS old, name_new AS new, created_at").order("created_at ASC"),
    @user.race_changes.includes(:old_race, :new_race).order("users_race_changes.created_at ASC")
  ]
  
  @changes[0].each do |clan_change|
    [
      # race_change['old_clan'] !== clan_change.old_clan
      {:association => clan_change.old_clan, :name => 'old'},
      {:association => clan_change.new_clan, :name => 'new'}
    ].each do |type|
      if clan_change["clan_id_#{type[:name]}"].eql?(0)
        clan_change[type[:name]] = nil
      elsif type[:association].nil?
          clan_change[type[:name]] = clan_change["clan_id_#{type[:name]}"]
      else
          clan_change[type[:name]] = type[:association].name
      end
    end
  end
  
  @changes[2].each do |race_change|
    [
      # race_change['old_race'] !== race_change.old_race
      {:association => race_change.old_race, :name => 'old'},
      {:association => race_change.new_race, :name => 'new'}
    ].each do |type|
      if race_change["race_id_#{type[:name]}"].eql?(0)
        race_change[type[:name]] = nil
      elsif type[:association].nil?
          race_change[type[:name]] = race_change["race_id_#{type[:name]}"]
      else
          race_change[type[:name]] = type[:association].name
      end
    end
  end
  
end

def list     
  @model = User
  @attributes = [
    {:db => "#{@model.table_name}.user_id", :human => "user_id"},
    {:db => "#{@model.table_name}.name", :human => "name"},
    {:db => "#{@model.table_name}.experience", :human => "experience"},
    {:db => "#{@model.table_name}.race_id", :human => "race"},
    {:db => "#{@model.table_name}.world_id", :human => "world"},
    {:db => "#{@model.table_name}.clan_id", :human => "clan"}     
  ]
  order = @attributes.select { |attr| attr[:human] == params[:order].downcase }.first
  order ||= @attributes[2]
  params[:order] = order[:human]
  
  @users = @model.includes(:world, :clan, :race).where(:world_id => @worlds).
           order("#{order[:db]} #{params[:by]}").offset(@offset).limit(@limit)   
  
  unless params[:name].nil?
    @users = @users.where("#{@model.table_name}.name LIKE ?", "%#{params[:name]}%") 
    @model = @model.where("#{@model.table_name}.name LIKE ?", "%#{params[:name]}%") 
  end    
  
  respond_to do |format|
    format.json { render :json => @users.limit(@suggest_limit) }
    format.jpeg { render_image('jpeg', [
      {:attr => 'user_id', :left => 0},
      {:attr => 'name', :left => 0.15},
      {:attr => 'experience', :left => 0.4},
      {:attr => 'race', :left => 0.6},
      {:attr => 'world_name', :left => 0.9}
    ])}
    format.html {render 'users/list'}
  end
end
